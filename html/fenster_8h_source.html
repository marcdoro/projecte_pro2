171static LRESULT CALLBACK fenster_wndproc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
172 struct fenster *f = (struct fenster *)GetWindowLongPtr(hwnd, GWLP_USERDATA);
173 switch (msg) {
174 case WM_PAINT: {
175 PAINTSTRUCT ps;
176 HDC hdc = BeginPaint(hwnd, &ps);
177 HDC memdc = CreateCompatibleDC(hdc);
178 HBITMAP hbmp = CreateCompatibleBitmap(hdc, f->width, f->height);
179 HBITMAP oldbmp = (HBITMAP)SelectObject(memdc, hbmp);
180 BINFO bi = {{sizeof(bi), f->width, -f->height, 1, 32, BI_BITFIELDS}};
181 bi.bmiColors[0].rgbRed = 0xff;
182 bi.bmiColors[1].rgbGreen = 0xff;
183 bi.bmiColors[2].rgbBlue = 0xff;
184 SetDIBitsToDevice(memdc, 0, 0, f->width, f->height, 0, 0, 0, f->height, f->buf,
185 (BITMAPINFO *)&bi, DIB_RGB_COLORS);
186 BitBlt(hdc, 0, 0, f->width, f->height, memdc, 0, 0, SRCCOPY);
187 SelectObject(memdc, oldbmp);
188 DeleteObject(hbmp);
189 DeleteDC(memdc);
190 EndPaint(hwnd, &ps);
191 } break;
192 case WM_CLOSE:
193 DestroyWindow(hwnd);
194 break;
195 case WM_LBUTTONDOWN:
196 case WM_LBUTTONUP:
197 f->mouse = (msg == WM_LBUTTONDOWN);
198 break;
199 case WM_MOUSEMOVE:
200 f->y = HIWORD(lParam), f->x = LOWORD(lParam);
201 break;
202 case WM_KEYDOWN:
203 case WM_KEYUP: {
204 f->mod = ((GetKeyState(VK_CONTROL) & 0x8000) >> 15) |
205 ((GetKeyState(VK_SHIFT) & 0x8000) >> 14) |
206 ((GetKeyState(VK_MENU) & 0x8000) >> 13) |
207 (((GetKeyState(VK_LWIN) | GetKeyState(VK_RWIN)) & 0x8000) >> 12);
208 f->keys[FENSTER_KEYCODES[HIWORD(lParam) & 0x1ff]] = !((lParam >> 31) & 1);
209 } break;
210 case WM_DESTROY:
211 PostQuitMessage(0);
212 break;
213 default:
214 return DefWindowProc(hwnd, msg, wParam, lParam);
215 }
216 return 0;
217}

286FENSTER_API int fenster_loop(struct fenster *f) {
287 XEvent ev;
288 XPutImage(f->dpy, f->w, f->gc, f->img, 0, 0, 0, 0, f->width, f->height);
289 XFlush(f->dpy);
290 while (XPending(f->dpy)) {
291 XNextEvent(f->dpy, &ev);
292 switch (ev.type) {
293 case ButtonPress:
294 case ButtonRelease:
295 f->mouse = (ev.type == ButtonPress);
296 break;
297 case MotionNotify:
298 f->x = ev.xmotion.x, f->y = ev.xmotion.y;
299 break;
300 case KeyPress:
301 case KeyRelease: {
302 int m = ev.xkey.state;
303 int k = XkbKeycodeToKeysym(f->dpy, ev.xkey.keycode, 0, 0);
304 for (unsigned int i = 0; i < 124; i += 2) {
305 if (FENSTER_KEYCODES[i] == k) {
306 f->keys[FENSTER_KEYCODES[i + 1]] = (ev.type == KeyPress);
307 break;
308 }
309 }
310 f->mod = (!!(m & ControlMask)) | (!!(m & ShiftMask) << 1) |
311 (!!(m & Mod1Mask) << 2) | (!!(m & Mod4Mask) << 3);
312 break;
313 }
314 case ClientMessage:
315 return -1;
316 }
317 }
318 return 0;
